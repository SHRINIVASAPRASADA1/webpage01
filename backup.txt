            <div class="faq-a">

                Any Indian citizen who has secured — or is in the process of
                securing — admission to a recognised institution in India or
                abroad.
              </p>
            </div>
          </div>
          <div class="faq-item">
            <div class="faq-q">
              What is the minimum CIBIL score needed?<span class="plus"></span>
            </div>
            <div class="faq-a">
              <p>
                Most lenders expect a co-applicant CIBIL score of at least 700
                for smooth processing. A score above 750 improves both approval
                speed and the rate offered. Below 680, select NBFCs still
                consider applications based on income stability.
              </p>
            </div>
          </div>
          <div class="faq-item">
            <div class="faq-q">
              Can I get a loan without collateral?<span class="plus"></span>
            </div>
            <div class="faq-a">
              <p>
                Yes. Several NBFCs and specialised lenders offer unsecured loans
                up to ₹40–75 Lakh depending on university ranking and course
                employability, generally at a 1.5–3% higher rate than secured
                loans.
              </p>
            </div>
          </div>
          <div class="faq-item">
            <div class="faq-q">
              How long does approval actually take?<span class="plus"></span>
            </div>
            <div class="faq-a">
              <p>
                Digital-first NBFCs typically sanction in 7–15 working days for
                clean files. Public sector banks usually take 4–6 weeks. Add 2–3
                weeks for disbursement coordination with your university.
              </p>
            </div>
          </div>
          <div class="faq-item">
            <div class="faq-q">
              What expenses does the loan cover?<span class="plus"></span>
            </div>
            <div class="faq-a">
              <p>
                Tuition, examination fees, living expenses, travel, insurance,
                study equipment, and pre-admission costs — disbursed in tranches
                aligned to your academic calendar, often direct to the
                university in foreign currency.
              </p>
            </div>
          </div>
          <div class="faq-item">
            <div class="faq-q">
              Who can be my co-applicant?<span class="plus"></span>
            </div>
            <div class="faq-a">
              <p>
                Typically a parent, sibling, parent-in-law or spouse — and with
                certain NBFC partners, a wider circle of relatives than banks
                usually permit.
              </p>
            </div>
          </div>
          <div class="faq-item">
            <div class="faq-q">
              Do I repay while I'm still studying?<span class="plus"></span>
            </div>
            <div class="faq-a">
              <p>

    <script>
      gsap.registerPlugin(ScrollTrigger, MotionPathPlugin);

      /* ---------- nav scroll state ---------- */
      const nav = document.getElementById("siteNav");
      ScrollTrigger.create({
        start: 60,
        onUpdate: (self) => {
          if (window.scrollY > 40) nav.classList.add("scrolled");
          else nav.classList.remove("scrolled");
        },
      });

      /* ---------- hero parallax layers ---------- */
      document.querySelectorAll(".hero-layer").forEach((layer) => {
        const speed = parseFloat(layer.dataset.speed || 0.2);
        gsap.to(layer, {
          yPercent: speed * 100,
          ease: "none",
          scrollTrigger: {
            trigger: ".hero",
            start: "top top",
            end: "bottom top",
            scrub: true,
          },
        });
      });

      /* hero plane gentle float */
      gsap.to(".hero-plane", {
        y: -14,
        x: 10,
        duration: 3.2,
        repeat: -1,
        yoyo: true,
        ease: "sine.inOut",
      });

      /* ---------- generic reveal on scroll ---------- */
      gsap.utils.toArray(".reveal").forEach((el) => {
        gsap.to(el, {
          opacity: 1,
          y: 0,
          duration: 0.9,
          ease: "power3.out",
          scrollTrigger: { trigger: el, start: "top 88%" },
        });
      });

      /* feature cards / compare / dest / testi / ov staggered even without .reveal class safety */
      gsap.utils
        .toArray(
          ".feat-card, .compare-card, .dest-card, .testi-card, .ov-card, .doc-item",
        )
        .forEach((el, i) => {
          gsap.fromTo(
            el,
            { opacity: 0, y: 26 },
            {
              opacity: 1,
              y: 0,
              duration: 0.7,
              ease: "power2.out",
              delay: (i % 3) * 0.06,
              scrollTrigger: { trigger: el, start: "top 92%" },
            },
          );
        });

      /* ---------- stat counters ---------- */
      document.querySelectorAll(".stat-num").forEach((el) => {
        const target = parseFloat(el.dataset.count);
        const suffix = el.dataset.suffix || "";
        const isDecimal = el.dataset.decimal === "true";
        let obj = { val: 0 };
        ScrollTrigger.create({
          trigger: el,
          start: "top 90%",
          once: true,
          onEnter: () => {
            gsap.to(obj, {
              val: target,
              duration: 1.8,
              ease: "power2.out",
              onUpdate: () => {
                el.textContent =
                  (isDecimal
                    ? obj.val.toFixed(1)
                    : Math.round(obj.val).toLocaleString("en-IN")) + suffix;
              },
            });
          },
        });
      });

      /* ---------- journey line draw + plane along path ---------- */
      ScrollTrigger.create({
        trigger: ".journey-path-wrap",
        start: "top 70%",
        end: "bottom 60%",
        scrub: 1,
        onUpdate: (self) => {
          const line = document.getElementById("journeyFg");
          const offset = 100 - self.progress * 100;
          line.setAttribute("stroke-dashoffset", offset);
          const plane = document.getElementById("journeyPlane");
          const wrap = document.querySelector(".journey-path-wrap");
          const h = wrap.offsetHeight;
          plane.style.top = self.progress * h + "px";
        },
      });

      /* journey cards reveal */
      gsap.utils.toArray(".j-card").forEach((card) => {
        gsap.to(card, {
          opacity: 1,
          y: 0,
          duration: 0.7,
          ease: "power2.out",
          scrollTrigger: { trigger: card, start: "top 85%" },
        });
      });

      /* ---------- FAQ accordion ---------- */
      document.querySelectorAll(".faq-item").forEach((item) => {
        const q = item.querySelector(".faq-q");
        const a = item.querySelector(".faq-a");
        q.addEventListener("click", () => {
          const isOpen = item.classList.contains("open");
          document.querySelectorAll(".faq-item.open").forEach((o) => {
            o.classList.remove("open");
            o.querySelector(".faq-a").style.maxHeight = null;
          });
          if (!isOpen) {
            item.classList.add("open");
            a.style.maxHeight = a.scrollHeight + "px";
          }
        });
      });

      /* ---------- eligibility chip selection ---------- */
      document.querySelectorAll(".chip-set").forEach((group) => {
        group.addEventListener("click", (e) => {
          if (!e.target.classList.contains("chip")) return;
          group
            .querySelectorAll(".chip")
            .forEach((c) => c.classList.remove("active"));
          e.target.classList.add("active");
        });
      });

      document.getElementById("eligBtn").addEventListener("click", () => {
        const cibil = document.querySelector(
          '[data-group="cibil"] .chip.active',
        );
        const collateral = document.querySelector(
          '[data-group="collateral"] .chip.active',
        );
        const admit = document.querySelector(
          '[data-group="admit"] .chip.active',
        );
        const result = document.getElementById("elig-result");

        if (!cibil || !collateral || !admit) {
          result.style.display = "block";
          result.innerHTML =
            "Please select an option in every field so we can give you an accurate read.";
          return;
        }

        let msg = "";
        const cibilText = cibil.textContent;
        const collateralYes = collateral.textContent === "Yes";

        if (collateralYes) {
          msg = `<strong>Strong fit for a secured loan.</strong> With collateral on the table, public sector banks are likely to offer you the most competitive rate — typically 8–11% — alongside a longer repayment tenure.`;
        } else if (cibilText === "750+") {
          msg = `<strong>Strong fit for a collateral-free NBFC loan.</strong> A co-applicant CIBIL score of ${cibilText} puts you in a strong position for fast, unsecured approval — often within 7–15 days.`;
        } else if (cibilText === "700–750") {
          msg = `<strong>Good fit for an unsecured loan.</strong> Most NBFC partners work comfortably in this CIBIL range. Expect a slightly higher rate than the lowest bracket, still with a fully digital process.`;
        } else if (cibilText === "Below 700") {
          msg = `<strong>Still fundable.</strong> A handful of our NBFC partners look beyond the CIBIL score to income stability and employment history — an advisor can confirm which lenders fit your profile.`;
        } else {
          msg = `<strong>Let's find out together.</strong> Share your co-applicant's PAN with our advisor for a free, no-obligation CIBIL pull before you apply anywhere.`;
        }

        if (admit.textContent === "Still applying") {
          msg += ` Several lenders also offer <strong>pre-admission, in-principle approval</strong> — useful while your applications are still in motion.`;
        }

        result.style.display = "block";
        result.innerHTML = msg;
      });

      /* ---------- lead form submit (demo) ---------- */
      document.getElementById("leadForm").addEventListener("submit", (e) => {
        e.preventDefault();
        const btn = e.target.querySelector('button[type="submit"]');
        const original = btn.textContent;
        btn.textContent = "Request received ✓";
        btn.style.opacity = 0.85;
        setTimeout(() => {
          btn.textContent = original;
          btn.style.opacity = 1;
          e.target.reset();
        }, 2400);
      });
    </script>
